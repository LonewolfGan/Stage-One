      })

Crée le hook apps/web/hooks/useSlotSync.ts :

  export function useSlotSync(providerId: string, date: string) {
    const queryClient = useQueryClient()
    useEffect(() => {
      const socket = getSocket()
      socket.connect()
      socket.emit('subscribe', { providerId, date })
      socket.on('slot.update', ({ slotStart, staffId, change }) => {
        // Invalider le cache React Query des slots
        queryClient.invalidateQueries({
          queryKey: ['slots', providerId, date]
        })
      })
      return () => {
        socket.off('slot.update')
        socket.disconnect()
      }
    }, [providerId, date])
  }

  import { Queue, Worker } from 'bullmq'

  export const expiryQueue = new Queue('booking-expiry', {
    connection: redis
  })

  new Worker('booking-expiry', async () => {
    const expired = await db
      .update(bookings)
      .set({ status: 'EXPIRED' })
      .where(
        and(
          eq(bookings.status, 'PENDING'),
          lt(bookings.lockedUntil, new Date())

          slotStart: booking.startDatetime,
          change: 'released'
        })

  }, { connection: redis })

PARTIE B — Contrainte GIST PostgreSQL

Exécute cette migration SQL directement sur la DB :

  CREATE EXTENSION IF NOT EXISTS btree_gist;

  ALTER TABLE bookings
    ADD CONSTRAINT no_overlapping_bookings
    EXCLUDE USING GIST (
      staff_id WITH =,
      tstzrange(start_datetime, end_datetime) WITH &&
    )
    WHERE (status NOT IN ('cancelled', 'expired'));

Ajoute cette migration dans un fichier

PARTIE C — SELECT FOR UPDATE dans la transaction

Dans la route POST /api/bookings, la transaction doit être :

  await db.transaction(async (tx) => {
    // 1. Vérifier les chevauchements avec lock de lecture
    const conflicts = await tx.execute(sql`
      SELECT id FROM bookings
      WHERE staff_id = ${staffId}
      AND status NOT IN ('cancelled', 'expired')
      AND tstzrange(start_datetime, end_datetime) &&
          tstzrange(${startDatetime}, ${endDatetime})
      FOR UPDATE
    `)
    if (conflicts.rows.length > 0) {
      throw new Error('SLOT_CONFLICT')
    }
    // 2. Insérer le booking
    const [booking] = await tx.insert(bookings).values({...}).returning()
    return booking
  })

  // Gérer l'erreur de conflit
  .catch((e) => {
    if (e.message === 'SLOT_CONFLICT'
        || e.code === 'P0001') {  // contrainte GIST
      return reply.status(409).send({ code: 'ERR-005' })
    }
    throw e
  })
