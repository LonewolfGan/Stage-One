CRITIQUE 1 — CONTRAINTE GIST POSTGRESQL

La contrainte anti-double-booking n'est pas en base.
Drizzle-kit ne la génère pas automatiquement.
Il faut l'appliquer via une migration SQL manuelle.

Crée le fichier :
apps/api/src/db/migrations/001_gist_constraint.sql

Contenu :
  CREATE EXTENSION IF NOT EXISTS btree_gist;

  DO $$
  BEGIN
    IF NOT EXISTS (
      SELECT 1 FROM pg_constraint
      WHERE conname = 'no_overlapping_bookings'
    ) THEN
      ALTER TABLE bookings
        ADD CONSTRAINT no_overlapping_bookings
        EXCLUDE USING GIST (
          staff_id WITH =,
          tstzrange(start_datetime, end_datetime) WITH &&
        )
        WHERE (status NOT IN ('cancelled', 'expired'));
    END IF;
  END $$;

Crée un script pour appliquer cette migration :
apps/api/src/db/run-migrations.ts

  import { readFileSync } from 'fs'
  import { join } from 'path'
  import { db } from './index'
  import { sql } from 'drizzle-orm'

  const migration = readFileSync(
    join(__dirname, 'migrations/001_gist_constraint.sql'),
    'utf8'
  )
  await db.execute(sql.raw(migration))
  console.log('Migration GIST appliquée avec succès')

Ajoute dans package.json de l'API :
  "scripts": {
    "migrate:gist": "tsx src/db/run-migrations.ts"
  }

Lance immédiatement :
  npm run migrate:gist

Vérifie que la contrainte est bien en base :
  SELECT conname FROM pg_constraint
  WHERE conname = 'no_overlapping_bookings'
  → doit retourner 1 ligne

apps/web/src/hooks/useSlotSync.ts :
  import { useEffect } from 'react'
  import { useQueryClient } from '@tanstack/react-query'
  import { getSocket } from '../lib/socket'

  export function useSlotSync(
    providerId: string,
    date: string,
    selectedSlot?: string,
    onConflict?: () => void
  ) {
    const queryClient = useQueryClient()
    useEffect(() => {
      if (!providerId) return
      const socket = getSocket()
      socket.connect()
      socket.emit('subscribe', { providerId, date })
      socket.on('slot.update', ({ slotStart, staffId, change }) => {
        queryClient.invalidateQueries({
          queryKey: ['slots', providerId, date]
        })
        if (
          change === 'booked' &&
          selectedSlot &&
          slotStart === selectedSlot &&
          onConflict
        ) {
          onConflict()
        }
      })
      return () => {
        socket.off('slot.update')
        socket.disconnect()
      }
    }, [providerId, date])
  }

apps/web/src/hooks/useBookingNotifications.ts :
  import { useEffect } from 'react'
  import { useQueryClient } from '@tanstack/react-query'
  import { toast } from 'sonner'
  import { getSocket } from '../lib/socket'
  import { format } from 'date-fns'
  import { fr } from 'date-fns/locale'

  export function useBookingNotifications(providerId: string) {
    const queryClient = useQueryClient()
    useEffect(() => {
      if (!providerId) return
      const socket = getSocket()
      socket.connect()
      socket.emit('subscribe', { providerId })
      socket.on('booking.confirmed', (booking) => {
        queryClient.invalidateQueries({
          queryKey: ['bookings', providerId]
        })
        toast.success(
          `Nouveau RDV — ${booking.clientName} · ${booking.serviceName} · ${
            format(new Date(booking.startDatetime), 'HH:mm', { locale: fr })
          }`
        )
      })
      return () => {
        socket.off('booking.confirmed')
        socket.disconnect()
      }
    }, [providerId])
  }

Middleware gating — crée requirePlan(plan: Plan) :
  export function requirePlan(minPlan: 'PRO' | 'BUSINESS') {
    return async (req, reply) => {
      const sub = await db.query.subscriptions.findFirst({
        where: eq(subscriptions.providerId, req.providerId)
      })
      const hierarchy = { FREE: 0, PRO: 1, BUSINESS: 2 }
      if (hierarchy[sub?.plan || 'FREE'] < hierarchy[minPlan]) {
        return reply.status(403).send({
          code: 'ERR-PLAN',
          message: `Cette fonctionnalité nécessite le plan ${minPlan}`
        })
      }
    }
  }
